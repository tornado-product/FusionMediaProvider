142    /// 从所有提供商搜索媒体
143    ///
144    /// 返回所有提供商的聚合结果，包含组合的分页信息
145    pub async fn search(&self, params: SearchParams) -> Result<AggregatedSearchResult> {
146        if self.providers.is_empty() {
147            return Err(MediaError::NoProviders);
148        }
149
150        let futures: Vec<_> = self.providers.iter().map(|provider| {
151            let provider = Arc::clone(provider);
152            let params = params.clone();
153
154            async move {
155                match params.media_type {
156                    MediaType::Image => {
157                        provider.search_images(&params.query, params.limit, params.page).await
158                    }
159                    MediaType::Video => {
160                        provider.search_videos(&params.query, params.limit, params.page).await
161                    }
162                }
163            }
164        }).collect();
165
166        let results = join_all(futures).await;
167
168        let mut provider_results = Vec::new();
169        let mut all_items = Vec::new();
170        let mut total_sum = 0u32;
171        let mut total_hits_sum = 0u32;
172        let mut total_pages_sum = 0u32;
173        let mut has_success = false;
174
175        for result in results {
176            match result {
177                Ok(search_result) => {
178                    has_success = true;
179
180                    // 聚合所有提供商的总数
181                    total_sum += search_result.total;
182                    total_hits_sum += search_result.total_hits;
183                    total_pages_sum += search_result.total_pages;
184
185                    // 收集项目
186                    all_items.extend(search_result.items.clone());
187
188                    // 存储提供商特定的结果
189                    provider_results.push(search_result);
190                }
191                Err(e) => {
192                    eprintln!("提供商失败: {}", e);
193                }
194            }
195        }
196
197        if !has_success {
198            return Err(MediaError::AllProvidersFailed);
199        }
200
201        Ok(AggregatedSearchResult {
202            provider: provider_results.first()
203                .map(|r| r.provider.clone())
204                .unwrap_or_else(|| "all".to_string()),
205            total: total_sum,
206            total_hits: total_hits_sum,
207            page: params.page,
208            per_page: params.limit,
209            total_pages: total_pages_sum,
210            items: all_items,
211            provider_results,
212        })
213    }

237    /// 下载单个媒体项并跟踪进度
238    pub async fn download_item(&self, item: &MediaItem) -> Result<String> {
239        let start_time = Instant::now();
240        let mut progress = DownloadProgress::new(item);
241
242        // 通知: 开始
243        progress.state = DownloadState::Starting;
244        self.notify_progress(&progress);
245
246        // 根据质量偏好确定 URL
247        let url = match item.media_type {
248            MediaType::Image => self.get_image_url(item)?,
249            MediaType::Video => self.get_video_url(item)?,
250        };
251
252        // 生成文件名
253        let filename = self.generate_filename(item);
254        let output_path = Path::new(&self.config.output_dir).join(&filename);
255
256        // 确保输出目录存在
257        tokio::fs::create_dir_all(&self.config.output_dir).await?;
258
259        // 开始下载
260        progress.state = DownloadState::Downloading;
261        self.notify_progress(&progress);
262
263        let response = self.http_client.get(&url).send().await?;
264
265        if !response.status().is_success() {
266            progress.state = DownloadState::Failed(
267                format!("HTTP {}", response.status())
268            );
269            self.notify_progress(&progress);
270            return Err(MediaError::DownloadError(
271                format!("HTTP {}: 下载失败", response.status())
272            ));
273        }
274
275        // 从 Content-Length 头获取总大小
276        progress.total_bytes = response.content_length();
277
278        // 下载并跟踪进度
279        let mut downloaded: u64 = 0;
280        let mut last_update = Instant::now();
281        let mut file = File::create(&output_path).await?;
282        let mut stream = response.bytes_stream();
283
284        use futures::StreamExt;
285        while let Some(chunk_result) = stream.next().await {
286            let chunk = chunk_result?;
287            let chunk_len = chunk.len() as u64;
288
289            // 写入块
290            file.write_all(&chunk).await?;
291            downloaded += chunk_len;
292
293            // 更新进度
294            let elapsed = start_time.elapsed().as_secs_f64();
295            progress.downloaded_bytes = downloaded;
296            progress.elapsed_secs = elapsed;
297            progress.speed_bps = if elapsed > 0.0 {
298                (downloaded as f64 / elapsed) as u64
299            } else {
300                0
301            };
302            progress.calculate_percentage();
303            progress.calculate_eta();
304
305            // 节流更新（每 100ms）
306            if last_update.elapsed().as_millis() >= 100 {
307                self.notify_progress(&progress);
308                last_update = Instant::now();
309            }
310        }
311
312        // 最终更新
313        progress.downloaded_bytes = downloaded;
314        progress.elapsed_secs = start_time.elapsed().as_secs_f64();
315        progress.calculate_percentage();
316
317        // 写入磁盘
318        progress.state = DownloadState::Writing;
319        self.notify_progress(&progress);
320
321        file.flush().await?;
322        drop(file);
323
324        // 完成
325        progress.state = DownloadState::Completed;
326        self.notify_progress(&progress);
327
328        Ok(output_path.to_string_lossy().to_string())
329    }

338    /// 并发批量下载多个媒体项，并跟踪整体进度
339    pub async fn download_items(&self, items: &[MediaItem]) -> Vec<Result<String>> {
340        let semaphore = Arc::new(tokio::sync::Semaphore::new(self.config.max_concurrent));
341
342        let futures: Vec<_> = items.iter().map(|item| {
343            let semaphore = Arc::clone(&semaphore);
344            let item = item.clone();
345            let downloader = self.clone();
346
347            async move {
348                let _permit = semaphore.acquire().await.unwrap();
349                downloader.download_item(&item).await
350            }
351        }).collect();
352
353        join_all(futures).await
354    }

356    /// Download multiple items with overall batch progress callback
357    pub async fn download_items_with_batch_progress<F>(
358        &self,
359        items: &[MediaItem],
360        batch_callback: F,
361    ) -> Vec<Result<String>>
362    where
363        F: Fn(BatchDownloadProgress) + Send + Sync + 'static,
364    {
365        use std::sync::Mutex;
366
367        let batch_progress = Arc::new(Mutex::new(BatchDownloadProgress::new(items.len())));
368        let batch_callback = Arc::new(batch_callback);
369
370        // Clone downloader with custom progress callback
371        let mut config = self.config.clone();
372        let batch_progress_clone = Arc::clone(&batch_progress);
373        let batch_callback_clone = Arc::clone(&batch_callback);
374
375        config.progress_callback = Some(Arc::new(move |progress: DownloadProgress| {
376            let mut batch = batch_progress_clone.lock().unwrap();
377
378            // Update item progress
379            if let Some(pos) = batch.item_progress.iter().position(|p| p.item_id == progress.item_id) {
380                batch.item_progress[pos] = progress.clone();
381            } else {
382                batch.item_progress.push(progress.clone());
383            }
384
385            // Update counters
386            match progress.state {
387                DownloadState::Downloading => {
388                    batch.downloading_items = batch.item_progress
389                        .iter()
390                        .filter(|p| matches!(p.state, DownloadState::Downloading))
391                        .count();
392                }
393                DownloadState::Completed => {
394                    batch.completed_items = batch.item_progress
395                        .iter()
396                        .filter(|p| matches!(p.state, DownloadState::Completed))
397                        .count();
398                }
399                DownloadState::Failed(_) => {
400                    batch.failed_items = batch.item_progress
401                        .iter()
402                        .filter(|p| matches!(p.state, DownloadState::Failed(_)))
403                        .count();
404                }
405                _ => {}
406            }
407
408            batch.calculate_overall_percentage();
409            batch_callback_clone(batch.clone());
410        }));
411
412        let downloader_with_callback = MediaDownloader {
413            providers: self.providers.clone(),
414            config,
415            http_client: self.http_client.clone(),
416        };
417
418        downloader_with_callback.download_items(items).await
419    }

421    /// 根据 ID 下载媒体
422    pub async fn download_by_id(&self, id: &str, media_type: MediaType) -> Result<String> {
423        // 遍历所有提供商尝试获取媒体
424        for provider in &self.providers {
425            match provider.get_media(id, media_type.clone()).await {
426                Ok(item) => {
427                    // 找到媒体项，下载它
428                    return self.download_item(&item).await;
429                }
430                Err(_) => {
431                    // 当前提供商没有找到，继续尝试下一个
432                    continue;
433                }
434            }
435        }
436
437        // 所有提供商都没有找到该媒体
438        Err(MediaError::DownloadError(format!("未找到 ID 为 {} 的媒体", id)))
439    }

441    /// 批量下载媒体项
442    pub async fn download_batch(
443        &self,
444        items: &[&MediaItem],
445        batch_callback: Option<ProgressCallback>,
446    ) -> Result<Vec<String>> {
447        let items_vec: Vec<MediaItem> = items.iter().map(|&item| item.clone()).collect();
448        let results = self.download_items_with_batch_progress(&items_vec, move |progress| {
449            if let Some(callback) = &batch_callback {
450                let progress = DownloadProgress {
451                    item_id: "batch".to_string(),
452                    item_title: format!("批量下载 ({}/{})", progress.completed_items, progress.total_items),
453                    provider: "聚合".to_string(),
454                    state: DownloadState::Completed,
455                    downloaded_bytes: 0,
456                    total_bytes: None,
457                    speed_bps: 0,
458                    percentage: progress.overall_percentage,
459                    elapsed_secs: 0.0,
460                    eta_secs: None,
461                };
462                callback(progress);
463            }
464        }).await;
465
466        let successful_downloads: Vec<String> = results
467            .into_iter()
468            .filter_map(|result| result.ok())
469            .collect();
470
471        Ok(successful_downloads)
472    }

474    /// 根据质量偏好获取图片 URL
475    fn get_image_url(&self, item: &MediaItem) -> Result<String> {
476        match self.config.image_quality {
477            ImageQuality::Thumbnail => Ok(item.urls.thumbnail.clone()),
478            ImageQuality::Medium => {
479                item.urls.medium.clone()
480                    .or_else(|| item.urls.large.clone())
481                    .or_else(|| Some(item.urls.thumbnail.clone()))
482                    .ok_or_else(|| MediaError::InvalidQuality("没有可用的中等质量".to_string()))
483            }
484            ImageQuality::Large => {
485                item.urls.large.clone()
486                    .or_else(|| item.urls.medium.clone())
487                    .or_else(|| Some(item.urls.thumbnail.clone()))
488                    .ok_or_else(|| MediaError::InvalidQuality("没有可用的大尺寸质量".to_string()))
489            }
490            ImageQuality::Original => {
491                item.urls.original.clone()
492                    .or_else(|| item.urls.large.clone())
493                    .or_else(|| item.urls.medium.clone())
494                    .or_else(|| Some(item.urls.thumbnail.clone()))
495                    .ok_or_else(|| MediaError::InvalidQuality("没有可用的原始质量".to_string()))
496            }
497        }
498    }

500    /// 根据质量偏好获取视频 URL
501    fn get_video_url(&self, item: &MediaItem) -> Result<String> {
502        let video_files = item.urls.video_files.as_ref()
503            .ok_or_else(|| MediaError::InvalidQuality("没有可用的视频文件".to_string()))?;
504
505        let quality_str = self.config.video_quality.as_str();
506
507        // 尝试查找精确的质量匹配
508        if let Some(file) = video_files.iter().find(|f| f.quality == quality_str) {
509            return Ok(file.url.clone());
510        }
511
512        // 尝试按分辨率查找
513        let min_width = self.config.video_quality.min_width();
514        if let Some(file) = video_files.iter()
515            .filter(|f| f.width >= min_width)
516            .min_by_key(|f| f.width)
517        {
518            return Ok(file.url.clone());
519        }
520
521        // 回退到最大可用
522        video_files.iter()
523            .max_by_key(|f| f.width)
524            .map(|f| f.url.clone())
525            .ok_or_else(|| MediaError::InvalidQuality("未找到合适的视频质量".to_string()))
526    }

528    /// 为媒体项生成文件名
529    fn generate_filename(&self, item: &MediaItem) -> String {
530        let extension = match item.media_type {
531            MediaType::Image => "jpg",
532            MediaType::Video => "mp4",
533        };
534
535        if self.config.use_original_names {
536            format!("{}_{}.{}", item.provider.to_lowercase(), item.id, extension)
537        } else {
538            let sanitized_title = item.title
539                .chars()
540                .filter(|c| c.is_alphanumeric() || *c == '_' || *c == '-')
541                .collect::<String>();
542
543            let sanitized = if sanitized_title.is_empty() {
544                item.id.clone()
545            } else {
546                sanitized_title
547            };
548
549            format!("{}_{}_{}.{}",
550                    item.provider.to_lowercase(),
551                    sanitized,
552                    item.id,
553                    extension
554            )
555        }
556    }
