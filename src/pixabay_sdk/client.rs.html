28    /// 在 Pixabay 上搜索图片
29    ///
30    /// # 参数
31    ///
32    /// * `query` - 搜索查询词（会自动进行 URL 编码，最多 100 字符）
33    /// * `per_page` - 每页结果数（3-200，默认：20）
34    /// * `page` - 页码（默认：1）
35    ///
36    /// # 可选参数（使用 SearchImageParams 构建器模式）
37    ///
38    /// 对于带有更多参数的高级搜索，请使用 `search_images_advanced()` 方法。
39    ///
40    /// # 速率限制
41    ///
42    /// 默认：每 60 秒 100 个请求（与 API 密钥关联，而非 IP 地址）
43    ///
44    /// # 示例
45    ///
46    /// ```no_run
47    /// # use pixabay_sdk::Pixabay;
48    /// # async fn example() -> Result<(), Box<dyn std::error::Error>> {
49    /// let client = Pixabay::new("your_api_key".to_string());
50    /// let images = client.search_images("yellow flowers", Some(10), Some(1)).await?;
51    /// println!("找到 {} 张图片", images.total_hits);
52    /// # Ok(())
53    /// # }
54    /// ```
55    pub async fn search_images(
56        &self,
57        query: &str,
58        per_page: Option<u32>,
59        page: Option<u32>,
60    ) -> Result<ImageResponse> {
61        // 验证 per_page 范围（根据 API 文档为 3-200）
62        let per_page = per_page.unwrap_or(20).clamp(3, 200);
63        let page = page.unwrap_or(1);
64
65        let mut url = Url::parse(BASE_URL)?;
66
67        url.query_pairs_mut()
68            .append_pair("key", &self.api_key)
69            .append_pair("q", query)
70            .append_pair("per_page", &per_page.to_string())
71            .append_pair("page", &page.to_string());
72
73        let response = self.client.get(url).send().await?;
74
75        self.handle_response(response).await
76    }

78    /// 处理 API 响应并提取相应的错误
79    async fn handle_response<T: serde::de::DeserializeOwned>(
80        &self,
81        response: reqwest::Response,
82    ) -> Result<T> {
83        let status = response.status();
84
85        if status.is_success() {
86            Ok(response.json().await?)
87        } else if status.as_u16() == 429 {
88            Err(PixabayError::RateLimitExceeded)
89        } else if status.as_u16() == 400 {
90            let error_text = response.text().await?;
91            Err(PixabayError::ApiError(format!("错误请求: {}", error_text)))
92        } else if status.as_u16() == 401 || status.as_u16() == 403 {
93            Err(PixabayError::InvalidApiKey)
94        } else {
95            let error_text = response.text().await?;
96            Err(PixabayError::ApiError(format!(
97                "HTTP {}: {}",
98                status.as_u16(),
99                error_text
100            )))
101        }
102    }

126    /// # Ok(())

129    pub async fn search_images_advanced(&self, params: SearchImageParams) -> Result<ImageResponse> {
130        // 验证 per_page 范围
131        let per_page = params.per_page.unwrap_or(20).clamp(3, 200);
132
133        let mut url = Url::parse(BASE_URL)?;
134
135        let mut query = url.query_pairs_mut();
136        query.append_pair("key", &self.api_key);
137
138        if let Some(q) = &params.query {
139            // 验证查询长度（根据 API 文档最多 100 字符）
140            if q.len() > 100 {
141                drop(query);
142                return Err(PixabayError::ApiError(
143                    "查询字符串不能超过 100 个字符".to_string(),
144                ));
145            }
146            query.append_pair("q", q);
147        }
148
149        query.append_pair("per_page", &per_page.to_string());
150        query.append_pair("page", &params.page.unwrap_or(1).to_string());
151
152        if let Some(image_type) = &params.image_type {
153            query.append_pair("image_type", &image_type.to_string());
154        }
155        if let Some(orientation) = &params.orientation {
156            query.append_pair("orientation", &orientation.to_string());
157        }
158        if let Some(category) = &params.category {
159            query.append_pair("category", &category.to_string());
160        }
161        if let Some(min_width) = params.min_width {
162            query.append_pair("min_width", &min_width.to_string());
163        }
164        if let Some(min_height) = params.min_height {
165            query.append_pair("min_height", &min_height.to_string());
166        }
167        if let Some(colors) = &params.colors {
168            query.append_pair("colors", colors);
169        }
170        if let Some(editors_choice) = params.editors_choice {
171            query.append_pair("editors_choice", &editors_choice.to_string());
172        }
173        if let Some(safesearch) = params.safesearch {
174            query.append_pair("safesearch", &safesearch.to_string());
175        }
176        if let Some(order) = &params.order {
177            query.append_pair("order", &order.to_string());
178        }
179        if let Some(lang) = &params.lang {
180            query.append_pair("lang", &lang.to_string());
181        }
182
183        drop(query);
184
185        let response = self.client.get(url).send().await?;
186        self.handle_response(response).await
187    }

206    pub async fn get_image(&self, id: u64) -> Result<Image> {
207        let mut url = Url::parse(BASE_URL)?;
208
209        url.query_pairs_mut()
210            .append_pair("key", &self.api_key)
211            .append_pair("id", &id.to_string());
212
213        let response = self.client.get(url).send().await?;
214
215        let image_response: ImageResponse = self.handle_response(response).await?;
216        image_response
217            .hits
218            .into_iter()
219            .next()
220            .ok_or_else(|| PixabayError::ApiError(format!("未找到 ID 为 {} 的图片", id)))
221    }
